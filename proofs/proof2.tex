\begin{question}
	If $AB$ is surjective as a transformation, then so is $A$.

	\begin{proof}
		\begin{align*}
			\forall z \in Z, \exists x \in X            & : (f\circ g)(x) = z \\
			f(g(x))                                     & = z                 \\
			\llet y\in Y                                & = g(x)              \\
			f(y)                              = f(g(x)) & = z                 \\
		\end{align*}
		$$ \therefore f \text{ is surjective.} $$
	\end{proof}
\end{question}
